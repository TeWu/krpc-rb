def encode_varint(value)
  return [value].pack('C') if value < 0b1000_0000
  result = ""
  loop do
    byte = value & 0b0111_1111
    value >>= 7
    if value == 0
      return result << byte.chr
    else
      result << (byte | 0b1000_0000).chr
    end
  end
end
